// backend/models/userModel.js
// File-based user model for Spetech Lost and Found Web
// - Uses centralized DB util (backend/lib/db.js) to avoid duplicate DB files
// - Provides user CRUD, authentication helpers, and seeding helper for init
// - Returns sanitized user objects (no password) for public responses

const bcrypt = require('bcrypt');
const { v4: uuidv4 } = require('uuid');

const { readDb, writeDb } = require('../lib/db');

const SALT_ROUNDS = 10;

/* --- Internal helpers (use centralized DB util) --- */

async function _readDb() {
  return await readDb();
}

async function _writeDb(db) {
  return await writeDb(db);
}

/* --- Model API --- */

/**
 * createUser({ username, password, role })
 * - hashes password and stores user
 * - returns safe user object (without password)
 */
async function createUser({ username, password, role = 'user' } = {}) {
  if (!username || !password) throw new Error('Username and password are required');

  const db = await _readDb();
  db.users = db.users || [];

  const exists = db.users.find(u => u.username.toLowerCase() === String(username).toLowerCase());
  if (exists) throw new Error('Username already exists');

  const hashed = await bcrypt.hash(String(password), SALT_ROUNDS);
  const user = {
    id: uuidv4(),
    username: String(username).trim(),
    password: hashed,
    role: String(role).trim(),
    createdAt: new Date().toISOString(),
    updatedAt: new Date().toISOString()
  };

  db.users.push(user);
  await _writeDb(db);

  return { id: user.id, username: user.username, role: user.role, createdAt: user.createdAt };
}

/**
 * findUserByUsername(username)
 * - returns full user object (including hashed password) or null
 */
async function findUserByUsername(username) {
  if (!username) return null;
  const db = await _readDb();
  db.users = db.users || [];
  return db.users.find(u => u.username.toLowerCase() === String(username).toLowerCase()) || null;
}

/**
 * getUserSafe(id)
 * - returns user object without password
 */
async function getUserSafe(id) {
  const db = await _readDb();
  db.users = db.users || [];
  const u = db.users.find(x => x.id === id);
  if (!u) return null;
  return { id: u.id, username: u.username, role: u.role, createdAt: u.createdAt };
}

/**
 * authenticateUser(username, password)
 * - verifies password, returns safe user on success, null on failure
 */
async function authenticateUser(username, password) {
  const user = await findUserByUsername(username);
  if (!user) return null;
  const match = await bcrypt.compare(String(password), user.password);
  if (!match) return null;
  return { id: user.id, username: user.username, role: user.role, createdAt: user.createdAt };
}

/**
 * listUsers({ includePasswords = false })
 * - returns array of users; by default hides password
 */
async function listUsers(opts = {}) {
  const { includePasswords = false } = opts;
  const db = await _readDb();
  db.users = db.users || [];
  if (includePasswords) return db.users.slice();
  return db.users.map(u => ({ id: u.id, username: u.username, role: u.role, createdAt: u.createdAt }));
}

/**
 * updateUserRole(id, newRole)
 * - updates role field
 */
async function updateUserRole(id, newRole) {
  if (!id || !newRole) throw new Error('id and newRole required');
  const db = await _readDb();
  db.users = db.users || [];
  const idx = db.users.findIndex(u => u.id === id);
  if (idx === -1) return null;
  db.users[idx].role = String(newRole).trim();
  db.users[idx].updatedAt = new Date().toISOString();
  await _writeDb(db);
  return { id: db.users[idx].id, username: db.users[idx].username, role: db.users[idx].role };
}

/**
 * deleteUser(id)
 * - removes user from DB (prototype behavior)
 */
async function deleteUser(id) {
  if (!id) throw new Error('id required');
  const db = await _readDb();
  db.users = db.users || [];
  const idx = db.users.findIndex(u => u.id === id);
  if (idx === -1) return false;
  db.users.splice(idx, 1);
  await _writeDb(db);
  return true;
}

/* --- New: seed default users for init (idempotent) --- */
/**
 * seedDefaultUsers({ adminPassword, userPassword })
 * - creates default admin and user only if missing
 * - returns array of created safe users (may be empty if already present)
 */
async function seedDefaultUsers({ adminPassword = 'admin123', userPassword = 'user123' } = {}) {
  // Use createUser so writes go through centralized DB util and validation
  const created = [];
  try {
    const db = await _readDb();
    db.users = db.users || [];

    const hasAdmin = db.users.some(u => u.username.toLowerCase() === 'admin');
    const hasUser = db.users.some(u => u.username.toLowerCase() === 'user');

    if (!hasAdmin) {
      try {
        const admin = await createUser({ username: 'admin', password: adminPassword, role: 'admin' });
        created.push(admin);
      } catch (e) {
        // ignore race or already exists
        console.warn('seedDefaultUsers: admin create skipped:', e.message);
      }
    }

    if (!hasUser) {
      try {
        const user = await createUser({ username: 'user', password: userPassword, role: 'user' });
        created.push(user);
      } catch (e) {
        console.warn('seedDefaultUsers: user create skipped:', e.message);
      }
    }
  } catch (err) {
    console.warn('seedDefaultUsers error', err);
  }
  return created;
}

/* --- Export --- */
module.exports = {
  createUser,
  findUserByUsername,
  getUserSafe,
  authenticateUser,
  listUsers,
  updateUserRole,
  deleteUser,
  seedDefaultUsers
};
