
/**
 * server.js
 * Minimal Node.js + Express backend for Spetech Lost and Found Web (prototype -> real-ready)
 *
 * Notes:
 * - Lightweight file-based "database" (db.json) for simplicity and reproducibility.
 * - Uses multer for file uploads (found photos) and stores them in ./uploads.
 * - Passwords hashed with bcrypt.
 * - This implementation focuses on clarity, modularity, and being runnable for KTI.
 * - For production: replace file DB with real RDBMS, add proper auth (JWT/sessions), input sanitization, rate-limiting.
 *
 * Endpoints (matching frontend expectations):
 *  - POST  /api/lost/add           {name, description, location, contact, type:'lost'}
 *  - GET   /api/lost/list
 *  - GET   /api/lost/recent
 *  - POST  /api/found/add          multipart/form-data (photo optional)
 *  - GET   /api/found/list
 *  - GET   /api/found/recent
 *  - POST  /api/account/register   {username, password, role}
 *  - POST  /api/account/login      {username, password}
 *  - GET   /api/admin/list?status=
 *  - POST  /api/admin/create-account {username,password,role}
 *  - DELETE /api/admin/delete/:id
 *  - PATCH  /api/admin/status/:id  {status}
 *
 * Run:
 *   npm install express multer bcrypt uuid fs-extra
 *   node server.js
 */

const express = require('express');
const path = require('path');
const fs = require('fs-extra');
const multer = require('multer');
const bcrypt = require('bcrypt');
const { v4: uuidv4 } = require('uuid');

const APP_ROOT = path.resolve(__dirname);
const FRONTEND_DIR = path.join(APP_ROOT, 'frontend');
const UPLOADS_DIR = path.join(APP_ROOT, 'uploads');
const DB_FILE = path.join(APP_ROOT, 'database', 'db.json');

const PORT = process.env.PORT || 3000;
const SALT_ROUNDS = 10;

// Ensure folders exist
fs.ensureDirSync(UPLOADS_DIR);
fs.ensureDirSync(path.dirname(DB_FILE));

// Initialize DB if missing
async function initDb() {
  const exists = await fs.pathExists(DB_FILE);
  if (!exists) {
    const initial = {
      users: [
        // default admin for convenience (password: admin123) - hashed below
      ],
      items: [] // items: { id, name, description, location, photo, contact, type: 'lost'|'found', status, createdAt, updatedAt }
    };
    // create default admin
    const hashed = await bcrypt.hash('admin123', SALT_ROUNDS);
    initial.users.push({ id: uuidv4(), username: 'admin', password: hashed, role: 'admin', createdAt: new Date().toISOString() });
    await fs.writeJson(DB_FILE, initial, { spaces: 2 });
    console.log('Initialized DB with default admin (username: admin, password: admin123)');
  }
}
initDb().catch(err => {
  console.error('DB init error', err);
  process.exit(1);
});

// Simple DB helpers (file-based)
async function readDb() {
  try {
    const data = await fs.readJson(DB_FILE);
    return data;
  } catch (err) {
    console.error('readDb error', err);
    return { users: [], items: [] };
  }
}
async function writeDb(data) {
  await fs.writeJson(DB_FILE, data, { spaces: 2 });
}

// Multer setup for file uploads
const storage = multer.diskStorage({
  destination: (req, file, cb) => cb(null, UPLOADS_DIR),
  filename: (req, file, cb) => {
    const ext = path.extname(file.originalname) || '';
    cb(null, `${Date.now()}-${uuidv4()}${ext}`);
  }
});
const upload = multer({ storage, limits: { fileSize: 5 * 1024 * 1024 } }); // 5MB limit

// Express app
const app = express();

// Middleware
app.use(express.json());
app.use(express.urlencoded({ extended: true }));

// Serve frontend static
app.use('/', express.static(FRONTEND_DIR));
app.use('/uploads', express.static(UPLOADS_DIR));

// Basic API router
const api = express.Router();
app.use('/api', api);

/* -------------------------
   Account endpoints
   ------------------------- */

// Register
api.post('/account/register', async (req, res) => {
  try {
    const { username, password, role = 'user' } = req.body;
    if (!username || !password) return res.json({ ok: false, message: 'Username dan password wajib.' });

    const db = await readDb();
    const exists = db.users.find(u => u.username.toLowerCase() === username.toLowerCase());
    if (exists) return res.json({ ok: false, message: 'Username sudah terdaftar.' });

    const hashed = await bcrypt.hash(password, SALT_ROUNDS);
    const user = { id: uuidv4(), username, password: hashed, role, createdAt: new Date().toISOString() };
    db.users.push(user);
    await writeDb(db);
    return res.json({ ok: true, message: 'Registrasi berhasil.' });
  } catch (err) {
    console.error('register error', err);
    return res.status(500).json({ ok: false, message: 'Server error saat registrasi.' });
  }
});

// Login
api.post('/account/login', async (req, res) => {
  try {
    const { username, password } = req.body;
    if (!username || !password) return res.json({ ok: false, message: 'Username dan password wajib.' });

    const db = await readDb();
    const user = db.users.find(u => u.username.toLowerCase() === username.toLowerCase());
    if (!user) return res.json({ ok: false, message: 'User tidak ditemukan.' });

    const match = await bcrypt.compare(password, user.password);
    if (!match) return res.json({ ok: false, message: 'Password salah.' });

    // Return user object without password
    const safeUser = { id: user.id, username: user.username, role: user.role };
    return res.json({ ok: true, user: safeUser });
  } catch (err) {
    console.error('login error', err);
    return res.status(500).json({ ok: false, message: 'Server error saat login.' });
  }
});

/* -------------------------
   Lost endpoints
   ------------------------- */

// Add lost report
api.post('/lost/add', async (req, res) => {
  try {
    const { name, description, location, contact } = req.body;
    if (!name || !description || !location) return res.status(400).json({ ok: false, message: 'Field wajib belum lengkap.' });

    const db = await readDb();
    const item = {
      id: uuidv4(),
      name,
      description,
      location,
      contact: contact || '',
      photo: null,
      type: 'lost',
      status: 'pending',
      createdAt: new Date().toISOString(),
      updatedAt: new Date().toISOString()
    };
    db.items.unshift(item);
    await writeDb(db);
    return res.json({ ok: true, item });
  } catch (err) {
    console.error('lost/add error', err);
    return res.status(500).json({ ok: false, message: 'Server error saat menambah laporan kehilangan.' });
  }
});

// List lost items
api.get('/lost/list', async (req, res) => {
  try {
    const db = await readDb();
    const items = db.items.filter(i => i.type === 'lost' && i.status !== 'deleted');
    return res.json({ ok: true, items });
  } catch (err) {
    console.error('lost/list error', err);
    return res.status(500).json({ ok: false, message: 'Server error.' });
  }
});

// Recent lost (for home)
api.get('/lost/recent', async (req, res) => {
  try {
    const db = await readDb();
    const items = db.items.filter(i => i.type === 'lost' && i.status !== 'deleted').slice(0, 8);
    return res.json({ ok: true, items });
  } catch (err) {
    console.error('lost/recent error', err);
    return res.status(500).json({ ok: false, message: 'Server error.' });
  }
});

/* -------------------------
   Found endpoints
   ------------------------- */

// Add found report (with optional photo)
api.post('/found/add', upload.single('photo'), async (req, res) => {
  try {
    const { name, description, location } = req.body;
    if (!name || !description || !location) return res.status(400).json({ ok: false, message: 'Field wajib belum lengkap.' });

    const db = await readDb();
    const photo = req.file ? `/uploads/${path.basename(req.file.path)}` : null;
    const item = {
      id: uuidv4(),
      name,
      description,
      location,
      contact: null,
      photo,
      type: 'found',
      status: 'pending',
      createdAt: new Date().toISOString(),
      updatedAt: new Date().toISOString()
    };
    db.items.unshift(item);
    await writeDb(db);
    return res.json({ ok: true, item });
  } catch (err) {
    console.error('found/add error', err);
    return res.status(500).json({ ok: false, message: 'Server error saat menambah laporan penemuan.' });
  }
});

// List found items
api.get('/found/list', async (req, res) => {
  try {
    const db = await readDb();
    const items = db.items.filter(i => i.type === 'found' && i.status !== 'deleted');
    return res.json({ ok: true, items });
  } catch (err) {
    console.error('found/list error', err);
    return res.status(500).json({ ok: false, message: 'Server error.' });
  }
});

// Recent found (for home)
api.get('/found/recent', async (req, res) => {
  try {
    const db = await readDb();
    const items = db.items.filter(i => i.type === 'found' && i.status !== 'deleted').slice(0, 8);
    return res.json({ ok: true, items });
  } catch (err) {
    console.error('found/recent error', err);
    return res.status(500).json({ ok: false, message: 'Server error.' });
  }
});

/* -------------------------
   Admin endpoints (verification, completion, archive, management)
   Note: For prototype, endpoints are not strictly authenticated.
   In production, protect these with proper auth & role checks.
   ------------------------- */

// List items by status (pending, verified, archived, all)
api.get('/admin/list', async (req, res) => {
  try {
    const status = (req.query.status || 'all').toLowerCase();
    const db = await readDb();
    let items = db.items.filter(i => i.status !== 'deleted');
    if (status !== 'all') items = items.filter(i => (i.status || '').toLowerCase() === status);
    return res.json({ ok: true, items });
  } catch (err) {
    console.error('admin/list error', err);
    return res.status(500).json({ ok: false, message: 'Server error.' });
  }
});

// Create account (admin management)
api.post('/admin/create-account', async (req, res) => {
  try {
    const { username, password, role = 'user' } = req.body;
    if (!username || !password) return res.status(400).json({ ok: false, message: 'Username & password wajib.' });

    const db = await readDb();
    const exists = db.users.find(u => u.username.toLowerCase() === username.toLowerCase());
    if (exists) return res.json({ ok: false, message: 'Username sudah ada.' });

    const hashed = await bcrypt.hash(password, SALT_ROUNDS);
    const user = { id: uuidv4(), username, password: hashed, role, createdAt: new Date().toISOString() };
    db.users.push(user);
    await writeDb(db);
    return res.json({ ok: true, message: 'Akun dibuat.' });
  } catch (err) {
    console.error('admin/create-account error', err);
    return res.status(500).json({ ok: false, message: 'Server error.' });
  }
});

// Delete form (mark deleted)
api.delete('/admin/delete/:id', async (req, res) => {
  try {
    const id = req.params.id;
    const db = await readDb();
    const idx = db.items.findIndex(i => i.id === id);
    if (idx === -1) return res.status(404).json({ ok: false, message: 'Item tidak ditemukan.' });

    // mark deleted
    db.items[idx].status = 'deleted';
    db.items[idx].updatedAt = new Date().toISOString();
    await writeDb(db);
    return res.json({ ok: true, message: 'Item dihapus.' });
  } catch (err) {
    console.error('admin/delete error', err);
    return res.status(500).json({ ok: false, message: 'Server error.' });
  }
});

// Update status
api.patch('/admin/status/:id', async (req, res) => {
  try {
    const id = req.params.id;
    const { status } = req.body;
    if (!status) return res.status(400).json({ ok: false, message: 'Status wajib.' });

    const allowed = ['pending', 'verified', 'completed', 'archived', 'deleted'];
    if (!allowed.includes(status)) return res.status(400).json({ ok: false, message: 'Status tidak valid.' });

    const db = await readDb();
    const item = db.items.find(i => i.id === id);
    if (!item) return res.status(404).json({ ok: false, message: 'Item tidak ditemukan.' });

    item.status = status;
    item.updatedAt = new Date().toISOString();
    await writeDb(db);
    return res.json({ ok: true, message: 'Status diperbarui.', item });
  } catch (err) {
    console.error('admin/status error', err);
    return res.status(500).json({ ok: false, message: 'Server error.' });
  }
});

/* -------------------------
   Fallback / health
   ------------------------- */

api.get('/health', (req, res) => res.json({ ok: true, uptime: process.uptime() }));

/* -------------------------
   Start server
   ------------------------- */

app.listen(PORT, () => {
  console.log(`Spetech LNF server running on http://localhost:${PORT}`);
  console.log(`Serving frontend from ${FRONTEND_DIR}`);
});
