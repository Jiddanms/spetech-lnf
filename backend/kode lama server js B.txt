
/**
 * server.js
 * Minimal Node.js + Express backend for Spetech Lost and Found Web (prototype -> real-ready)
 *
 * Notes:
 * - Lightweight file-based "database" (db.json) for simplicity and reproducibility.
 * - Uses multer for file uploads (found photos) and stores them in ./uploads.
 * - Passwords hashed with bcrypt.
 * - This implementation focuses on clarity, modularity, and being runnable for KTI.
 * - For production: replace file DB with real RDBMS, add proper auth (JWT/sessions), input sanitization, rate-limiting.
 *
 * Endpoints (matching frontend expectations):
 *  - POST  /api/lost/add           {name, description, location, contact, type:'lost'}
 *  - GET   /api/lost/list
 *  - GET   /api/lost/recent
 *  - POST  /api/found/add          multipart/form-data (photo optional)
 *  - GET   /api/found/list
 *  - GET   /api/found/recent
 *  - POST  /api/account/register   {username, password, role}
 *  - POST  /api/account/login      {username, password}
 *  - GET   /api/admin/list?status=
 *  - POST  /api/admin/create-account {username,password,role}
 *  - DELETE /api/admin/delete/:id
 *  - PATCH  /api/admin/status/:id  {status}
 *
 * Run:
 *   npm install express multer bcrypt uuid fs-extra
 *   node server.js
 */

/**
 * backend/server.js
 * Updated for QoL: Admin Form Management & Account Management,
 * session-based login token, seed dummy accounts (admin + user).
 *
 * Prototype-level auth: Authorization: Bearer <token>
 *
 * Note: For production, replace session mechanism with secure JWT or server session store.
 */

const express = require('express');
const path = require('path');
const fs = require('fs-extra');
const multer = require('multer');
const bcrypt = require('bcrypt');
const { v4: uuidv4 } = require('uuid');

const userModel = require('./models/userModel');

const APP_ROOT = path.resolve(__dirname, '..');
const FRONTEND_DIR = path.join(APP_ROOT, 'frontend');
const UPLOADS_DIR = path.join(APP_ROOT, 'uploads');
const DB_FILE = path.join(APP_ROOT, 'database', 'db.json');

const PORT = process.env.PORT || 3000;
const SALT_ROUNDS = 10;

// Ensure folders exist
fs.ensureDirSync(UPLOADS_DIR);
fs.ensureDirSync(path.dirname(DB_FILE));

// Simple DB helpers (file-based)
async function readDb() {
  try {
    const data = await fs.readJson(DB_FILE);
    return data;
  } catch (err) {
    console.error('readDb error', err);
    return { users: [], items: [], sessions: [] };
  }
}
async function writeDb(data) {
  await fs.writeJson(DB_FILE, data, { spaces: 2 });
}

// Initialize DB if missing and seed dummy accounts for testing (admin + user)
async function initDb() {
  const exists = await fs.pathExists(DB_FILE);
  if (!exists) {
    const initial = {
      users: [],
      items: [],
      sessions: []
    };

    // create default admin (admin/admin123) and default user (user/user123)
    const hashedAdmin = await bcrypt.hash('admin123', SALT_ROUNDS);
    const hashedUser = await bcrypt.hash('user123', SALT_ROUNDS);

    initial.users.push({
      id: uuidv4(),
      username: 'admin',
      password: hashedAdmin,
      role: 'admin',
      createdAt: new Date().toISOString(),
      updatedAt: new Date().toISOString()
    });

    initial.users.push({
      id: uuidv4(),
      username: 'user',
      password: hashedUser,
      role: 'user',
      createdAt: new Date().toISOString(),
      updatedAt: new Date().toISOString()
    });

    await fs.writeJson(DB_FILE, initial, { spaces: 2 });
    console.log('Initialized DB with default accounts:');
    console.log(' - admin / admin123 (role: admin)');
    console.log(' - user  / user123  (role: user)');
  } else {
    // ensure sessions array exists if DB already present
    try {
      const db = await readDb();
      if (!Array.isArray(db.sessions)) {
        db.sessions = [];
        await writeDb(db);
      }
    } catch (e) {
      console.warn('initDb: unable to normalize sessions array', e);
    }
  }
}

// call initDb and then seed default users (idempotent)
initDb()
  .then(async () => {
    try {
      const created = await userModel.seedDefaultUsers();
      if (created && created.length) {
        console.log('Seeded default users:', created.map(u => u.username).join(', '));
      } else {
        console.log('Default users already present or none created.');
      }
    } catch (seedErr) {
      console.warn('Seeding default users failed:', seedErr);
    }
  })
  .catch(err => {
    console.error('DB init error', err);
    process.exit(1);
  });


// Multer setup for file uploads
const storage = multer.diskStorage({
  destination: (req, file, cb) => cb(null, UPLOADS_DIR),
  filename: (req, file, cb) => {
    const ext = path.extname(file.originalname) || '';
    cb(null, `${Date.now()}-${uuidv4()}${ext}`);
  }
});
const upload = multer({ storage, limits: { fileSize: 5 * 1024 * 1024 } }); // 5MB limit

// Express app
const app = express();

// Middleware
app.use(express.json());
app.use(express.urlencoded({ extended: true }));

// Serve frontend static
app.use('/', express.static(FRONTEND_DIR));
app.use('/uploads', express.static(UPLOADS_DIR));

// Basic API router
const api = express.Router();
app.use('/api', api);

/* -------------------------
   Simple auth middleware (prototype)
   - Expects Authorization: Bearer <token>
   - Validates session in db.sessions
   ------------------------- */
async function requireAuth(req, res, next) {
  try {
    const auth = req.headers['authorization'] || '';
    const parts = auth.split(' ');
    if (parts.length !== 2 || parts[0] !== 'Bearer') return res.status(401).json({ ok: false, message: 'Unauthorized' });
    const token = parts[1];
    const db = await readDb();
    const session = (db.sessions || []).find(s => s.token === token);
    if (!session) return res.status(401).json({ ok: false, message: 'Session tidak ditemukan' });
    if (new Date(session.expiresAt) < new Date()) return res.status(401).json({ ok: false, message: 'Session expired' });

    const user = (db.users || []).find(u => u.id === session.userId);
    if (!user) return res.status(401).json({ ok: false, message: 'User tidak ditemukan' });

    // attach user to request
    req.user = { id: user.id, username: user.username, role: user.role };
    next();
  } catch (err) {
    console.error('requireAuth error', err);
    return res.status(500).json({ ok: false, message: 'Auth error' });
  }
}

function requireAdmin(req, res, next) {
  if (!req.user) return res.status(401).json({ ok: false, message: 'Unauthorized' });
  if (req.user.role !== 'admin') return res.status(403).json({ ok: false, message: 'Admin only' });
  next();
}

   Account endpoints

// Register
api.post('/account/register', async (req, res) => {
  try {
    const { username, password, role = 'user' } = req.body;
    if (!username || !password) return res.status(400).json({ ok: false, message: 'Username dan password wajib.' });

    const db = await readDb();
    const exists = db.users.find(u => u.username.toLowerCase() === username.toLowerCase());
    if (exists) return res.status(400).json({ ok: false, message: 'Username sudah terdaftar.' });

    const hashed = await bcrypt.hash(password, SALT_ROUNDS);
    const user = { id: uuidv4(), username, password: hashed, role, createdAt: new Date().toISOString(), updatedAt: new Date().toISOString() };
    db.users.push(user);
    await writeDb(db);
    return res.json({ ok: true, message: 'Registrasi berhasil.' });
  } catch (err) {
    console.error('register error', err);
    return res.status(500).json({ ok: false, message: 'Server error saat registrasi.' });
  }
});

// Login (creates a simple session token for prototype)
// Login v2 (menggunakan userModel.authenticateUser)
api.post('/account/login', async (req, res) => {
  try {
    const { username, password } = req.body;
    if (!username || !password) return res.status(400).json({ ok: false, message: 'Username dan password wajib.' });

    const authUser = await userModel.authenticateUser(username, password);
    if (!authUser) return res.status(401).json({ ok: false, message: 'Username atau password salah.' });

    // create simple session token (prototype)
    const db = await readDb();
    db.sessions = db.sessions || [];
    const token = uuidv4();
    const expiresAt = new Date(Date.now() + 24 * 60 * 60 * 1000).toISOString();
    db.sessions.push({ token, userId: authUser.id, createdAt: new Date().toISOString(), expiresAt });
    await writeDb(db);

    return res.json({ ok: true, user: { id: authUser.id, username: authUser.username, role: authUser.role }, token });
  } catch (err) {
    console.error('login error', err);
    return res.status(500).json({ ok: false, message: 'Server error saat login.' });
  }
});

// Logout (invalidate token)
api.post('/account/logout', requireAuth, async (req, res) => {
  try {
    const auth = req.headers['authorization'] || '';
    const token = auth.split(' ')[1];
    const db = await readDb();
    db.sessions = (db.sessions || []).filter(s => s.token !== token);
    await writeDb(db);
    return res.json({ ok: true, message: 'Logged out' });
  } catch (err) {
    console.error('logout error', err);
    return res.status(500).json({ ok: false, message: 'Server error.' });
  }
});

// Add lost report
api.post('/lost/add', async (req, res) => {
  try {
    const { name, description, location, contact } = req.body;
    if (!name || !description || !location) return res.status(400).json({ ok: false, message: 'Field wajib belum lengkap.' });

    const db = await readDb();
    const item = {
      id: uuidv4(),
      name,
      description,
      location,
      contact: contact || '',
      photo: null,
      type: 'lost',
      status: 'pending',
      createdAt: new Date().toISOString(),
      updatedAt: new Date().toISOString()
    };
    db.items.unshift(item);
    await writeDb(db);
    return res.json({ ok: true, item });
  } catch (err) {
    console.error('lost/add error', err);
    return res.status(500).json({ ok: false, message: 'Server error saat menambah laporan kehilangan.' });
  }
});

// List lost items

/* -------------------------
   Admin endpoints (Form Management & Account Management)
   Protected by requireAuth + requireAdmin
   ------------------------- */

const adminRouter = express.Router();
adminRouter.use(requireAuth, requireAdmin);

// FORMS: list all (optionally filter by type or status)
adminRouter.get('/forms', async (req, res) => {
  try {
    const type = req.query.type; // optional: 'lost' or 'found'
    const status = req.query.status; // optional
    const db = await readDb();
    let items = (db.items || []).filter(i => i.status !== 'deleted');
    if (type) items = items.filter(i => i.type === type);
    if (status) items = items.filter(i => i.status === status);
    return res.json({ ok: true, items });
  } catch (err) {
    console.error('GET /admin/forms error', err);
    return res.status(500).json({ ok: false, message: 'Server error.' });
  }
});

// Update item status
adminRouter.patch('/forms/:id/status', async (req, res) => {
  try {
    const id = req.params.id;
    const { status } = req.body;
    const allowed = ['pending', 'verified', 'completed', 'archived', 'deleted'];
    if (!status || !allowed.includes(status)) return res.status(400).json({ ok: false, message: 'Status tidak valid.' });

    const db = await readDb();
    const item = db.items.find(i => i.id === id);
    if (!item) return res.status(404).json({ ok: false, message: 'Item tidak ditemukan.' });

    item.status = status;
    item.updatedAt = new Date().toISOString();
    await writeDb(db);
    return res.json({ ok: true, item });
  } catch (err) {
    console.error('PATCH /admin/forms/:id/status error', err);
    return res.status(500).json({ ok: false, message: 'Server error.' });
  }
});

// Delete item (soft delete)
adminRouter.delete('/forms/:id', async (req, res) => {
  try {
    const id = req.params.id;
    const db = await readDb();
    const item = db.items.find(i => i.id === id);
    if (!item) return res.status(404).json({ ok: false, message: 'Item tidak ditemukan.' });
    item.status = 'deleted';
    item.updatedAt = new Date().toISOString();
    await writeDb(db);
    return res.json({ ok: true, message: 'Item dihapus.' });
  } catch (err) {
    console.error('DELETE /admin/forms/:id error', err);
    return res.status(500).json({ ok: false, message: 'Server error.' });
  }
});

adminRouter.get('/accounts', async (req, res) => {

  } catch (err) {
    console.error('GET /admin/accounts error', err);
    return res.status(500).json({ ok: false, message: 'Server error.' });

adminRouter.post('/accounts', async (req, res) => {
  try {
    const { username, password, role = 'user' } = req.body;
    if (!username || !password) return res.status(400).json({ ok: false, message: 'Username & password wajib.' });

    const db = await readDb();
    const exists = db.users.find(u => u.username.toLowerCase() === username.toLowerCase());
    if (exists) return res.status(400).json({ ok: false, message: 'Username sudah ada.' });

    const hashed = await bcrypt.hash(password, SALT_ROUNDS);
    const user = { id: uuidv4(), username, password: hashed, role, createdAt: new Date().toISOString(), updatedAt: new Date().toISOString() };
    db.users.push(user);
    await writeDb(db);
    return res.json({ ok: true, user: { id: user.id, username: user.username, role: user.role } });
  } catch (err) {
    console.error('POST /admin/accounts error', err);
    return res.status(500).json({ ok: false, message: 'Server error.' });
  }
});

// Update role
adminRouter.patch('/accounts/:id/role', async (req, res) => {
  try {
    const id = req.params.id;
    const { role } = req.body;
    if (!role || !['user', 'admin'].includes(role)) return res.status(400).json({ ok: false, message: 'Role tidak valid.' });

    const db = await readDb();
    const user = db.users.find(u => u.id === id);
    if (!user) return res.status(404).json({ ok: false, message: 'User tidak ditemukan.' });

    // Prevent admin from demoting themselves (safety)
    if (user.id === req.user.id && user.role === 'admin' && role !== 'admin') {
      return res.status(400).json({ ok: false, message: 'Tidak dapat mengubah role akun sendiri.' });
    }

    user.role = role;
    user.updatedAt = new Date().toISOString();
    await writeDb(db);
    return res.json({ ok: true, user: { id: user.id, username: user.username, role: user.role } });
  } catch (err) {
    console.error('PATCH /admin/accounts/:id/role error', err);
    return res.status(500).json({ ok: false, message: 'Server error.' });
  }
});

// Delete account
adminRouter.delete('/accounts/:id', async (req, res) => {
  try {
    const id = req.params.id;
    // Prevent deleting self
    if (id === req.user.id) return res.status(400).json({ ok: false, message: 'Tidak dapat menghapus akun sendiri.' });

    const db = await readDb();
    const idx = db.users.findIndex(u => u.id === id);
    if (idx === -1) return res.status(404).json({ ok: false, message: 'User tidak ditemukan.' });

    db.users.splice(idx, 1);
    // Also remove sessions for that user
    db.sessions = (db.sessions || []).filter(s => s.userId !== id);
    await writeDb(db);
    return res.json({ ok: true, message: 'User dihapus.' });
  } catch (err) {
    console.error('DELETE /admin/accounts/:id error', err);
    return res.status(500).json({ ok: false, message: 'Server error.' });
  }
});

// mount admin router under /admin
api.use('/admin', adminRouter);
